function getActionBtn(b) {
  if (role === 'engineer' && b.status === 'submitted') {
    return `<button class="btn btn-ghost" style="padding:3px 8px;font-size:11px" onclick="openApproval('${b.id}','verify')">Verify</button>`;
  }
  if (role === 'gm' && b.status === 'verified') {
    return `<button class="btn btn-ghost" style="padding:3px 8px;font-size:11px" onclick="openApproval('${b.id}','approve')">Approve</button>`;
  }
  if (role === 'owner' && (b.status === 'submitted' || b.status === 'verified')) {
    return `<button class="btn btn-ghost" style="padding:3px 8px;font-size:11px" onclick="openApproval('${b.id}','approve')">Review</button>`;
  }
  return '';
}

function renderBills() {
  const statuses = ['All','draft','submitted','verified','approved','rejected'];
  const fDiv = document.getElementById('billFilters');
  fDiv.innerHTML = statuses.map(s => `
    <div class="filter-chip ${billFilter===s?'active':''}" onclick="setBillFilter('${s}')">${s.charAt(0).toUpperCase()+s.slice(1)}</div>
  `).join('');

  const q = document.getElementById('billSearch')?.value?.toLowerCase() || '';
  let filtered = bills.filter(b => {
    const w = works.find(x => x.id === b.workId);
    const matchStatus = billFilter === 'All' || b.status === billFilter;
    const matchQ = !q || b.ref.toLowerCase().includes(q) || w?.name.toLowerCase().includes(q);
    // Role filter
    if (role === 'contractor') return matchStatus && matchQ;
    if (role === 'engineer') return matchStatus && matchQ && ['submitted','verified','approved'].includes(b.status);

function updateRateType() {
  const rt = document.getElementById('wi-rateType').value;
  const isLS = rt === 'lumpsum';
  document.getElementById('wi-qty-rate-row').style.display = isLS ? 'none' : '';
  document.getElementById('wi-lumpsum-row').style.display = isLS ? '' : 'none';
  const labels = { sqft:'Contracted Quantity (Sq.Ft.)', unit:'Contracted Quantity (Units)', rmt:'Contracted Quantity (RMT)', cum:'Contracted Quantity (CUM)' };
  if (!isLS) document.getElementById('wi-qty-label').textContent = labels[rt] || 'Contracted Quantity';
  calcWI();
}

function calcWI() {
  const rt = document.getElementById('wi-rateType').value;
  let value, qty, rate;
  if (rt === 'lumpsum') {
    value = parseFloat(document.getElementById('wi-lumpsum').value) || 0;
    qty = 1; rate = value;
  } else {
    qty  = parseFloat(document.getElementById('wi-qty').value) || 0;
    rate = parseFloat(document.getElementById('wi-rate').value) || 0;
    value = qty * rate;
  }
  if (!value) { document.getElementById('wi-calc').style.display='none'; return; }
  document.getElementById('wi-calc').style.display = '';
  document.getElementById('wc-qty').textContent   = rt==='lumpsum' ? 'Lump Sum' : qty.toLocaleString('en-IN')+' '+fmtUnit(rt);
  document.getElementById('wc-rate').textContent  = rt==='lumpsum' ? 'Lump Sum' : fmt(rate)+'/'+fmtUnit(rt);
  document.getElementById('wc-total').textContent = fmt(value);
  document.getElementById('wc-gst').textContent   = fmt(value*0.18);
  document.getElementById('wc-gross').textContent = fmt(value*1.18);
}

function saveWorkItem() {
  const name = document.getElementById('wi-name').value.trim();
  const rt   = document.getElementById('wi-rateType').value;
  if (!name) { showToast('⚠️','Enter work item name'); return; }

  let qty, rate;
  if (rt === 'lumpsum') {
    const ls = parseFloat(document.getElementById('wi-lumpsum').value);
    if (!ls) { showToast('⚠️','Enter lump sum value'); return; }
    qty = 1; rate = ls;
  } else {
    qty  = parseFloat(document.getElementById('wi-qty').value);
    rate = parseFloat(document.getElementById('wi-rate').value);
    if (!qty || !rate) { showToast('⚠️','Enter quantity and rate'); return; }
  }

  works.push({
    id: 'w' + (++nextWorkId),
    name,
    cat: document.getElementById('wi-cat').value,
    rateType: rt,
    qty, rate,
    contractor: document.getElementById('wi-contractor').value.trim() || 'TBD',
    desc: document.getElementById('wi-desc').value.trim(),
  });
  closeModal('addWorkModal');
  renderWorks();
  renderAll();
  showToast('✅', 'Work item added — opening ledger created');
}

function onBillWorkChange() {
  const wid = document.getElementById('bill-work').value;
  const info = document.getElementById('bill-remaining-info');
  if (!wid) { info.style.display='none'; return; }
  const w = works.find(x => x.id === wid);
  if (!w) return;
  const cert = certifiedQty(wid);
  const pend = pendingQty(wid);
  const rem  = w.qty - cert - pend;
  document.getElementById('br-contracted').textContent = w.qty.toLocaleString('en-IN')+' '+fmtUnit(w.rateType);
  document.getElementById('br-certified').textContent  = cert.toLocaleString('en-IN');
  document.getElementById('br-remaining').textContent  = rem.toLocaleString('en-IN');
  document.getElementById('bill-qty-label').textContent = `Qty This Bill (${fmtUnit(w.rateType)})`;
  info.style.display = '';
  calcBill();
}

function calcBill() {
  const wid = document.getElementById('bill-work').value;
  if (!wid) return;
  const w = works.find(x => x.id === wid);
  if (!w) return;
  const qty  = parseFloat(document.getElementById('bill-qty').value) || 0;
  if (!qty) { document.getElementById('bill-calc').style.display='none'; return; }
  const rate = w.rateType === 'lumpsum' ? w.rate / w.qty : w.rate;
  const base = qty * rate;
  const gst  = base * 0.18;
  const gross = base + gst;
  const cert  = certifiedQty(wid);
  const pend  = pendingQty(wid);
  const remaining = w.qty - cert - pend - qty;
  document.getElementById('bill-calc').style.display = '';
  document.getElementById('bc-qty').textContent    = qty.toLocaleString('en-IN')+' '+fmtUnit(w.rateType);
  document.getElementById('bc-rate').textContent   = fmt(rate)+'/'+fmtUnit(w.rateType);
  document.getElementById('bc-base').textContent   = fmt(base);
  document.getElementById('bc-gst').textContent    = fmt(gst);
  document.getElementById('bc-gross').textContent  = fmt(gross);
  document.getElementById('bc-remaining').textContent = remaining.toLocaleString('en-IN')+' '+fmtUnit(w.rateType)+(remaining<0?' ⚠️ EXCEEDS CONTRACT':'');
}

function saveBill() {
  const wid = document.getElementById('bill-work').value;
  const qty = parseFloat(document.getElementById('bill-qty').value);
  const ref = document.getElementById('bill-ref').value.trim();
  if (!wid) { showToast('⚠️','Select work item'); return; }
  if (!qty)  { showToast('⚠️','Enter quantity'); return; }

  const w = works.find(x => x.id === wid);
  const cert = certifiedQty(wid), pend = pendingQty(wid);
  if (qty > w.qty - cert - pend) {
    if (!confirm(`This quantity (${qty.toLocaleString('en-IN')}) exceeds the remaining balance. Submit anyway?`)) return;
  }

  bills.push({
    id: 'b' + (++nextBillId),
    workId: wid,
    ref: ref || ('BILL/' + nextBillId),
    qty,
    date: document.getElementById('bill-date').value,
    remarks: document.getElementById('bill-remarks').value.trim(),
    status: 'submitted',
    submittedAt: new Date().toISOString().slice(0,10),
  });
  closeModal('addBillModal');
  renderAll();
  showToast('✅', 'Bill submitted — awaiting site verification');
}

function openApproval(billId, action) {
  currentApprovalBillId = billId;
  approvalAction = action;

  const base = b.qty * (w.rateType === 'lumpsum' ? w.rate / w.qty : w.rate);

  const gross = base + gst;
  document.getElementById('appr-modal-title').textContent =

    action === 'approve' ? 'Approve Bill (GM / Owner)' : 'Reject Bill';
  document.getElementById('appr-modal-sub').textContent = `${w.name} · ${b.ref} · ${b.date}`;
  document.getElementById('appr-calc').innerHTML = `
    <div class="calc-row"><span>Work Item</span><span style="color:var(--text)">${w.name}</span></div>
    <div class="calc-row"><span>Qty</span><span style="color:var(--text);font-family:var(--mono)">${b.qty.toLocaleString('en-IN')} ${fmtUnit(w.rateType)}</span></div>
    <div class="calc-row"><span>Base Amount</span><span>${fmt(base)}</span></div>
    <div class="calc-row"><span>GST @ 18%</span><span>${fmt(gst)}</span></div>
    <div class="calc-row total"><span>GROSS AMOUNT</span><span>${fmt(gross)}</span></div>
    <div class="calc-row"><span>TDS @ 1%</span><span>(${fmt(tds)})</span></div>
    <div class="calc-row net"><span>NET PAYABLE</span><span>${fmt(gross - tds)}</span></div>
  `;
  document.getElementById('appr-remarks').value = '';
  const actDiv = document.getElementById('appr-actions');
  if (action === 'reject') {
    actDiv.innerHTML = `<button class="btn btn-ghost" onclick="closeModal('approveModal')">Cancel</button>
      <button class="btn btn-red" onclick="executeApproval('reject')">✗ Reject Bill</button>`;
  } else if (action === 'verify') {
    actDiv.innerHTML = `<button class="btn btn-ghost" onclick="closeModal('approveModal')">Cancel</button>
      <button class="btn btn-red" onclick="executeApproval('reject')">✗ Reject</button>
      <button class="btn btn-green" onclick="executeApproval('verify')">✓ Verify</button>`;
  } else {
    actDiv.innerHTML = `<button class="btn btn-ghost" onclick="closeModal('approveModal')">Cancel</button>
      <button class="btn btn-red" onclick="executeApproval('reject')">✗ Reject</button>
      <button class="btn btn-amber" onclick="executeApproval('approve')">✓ Approve & Certify</button>`;
  }
  openModal('approveModal');
}

function executeApproval(action) {
  const b = bills.find(x => x.id === currentApprovalBillId);
  if (!b) return;
  const remarks = document.getElementById('appr-remarks').value.trim();
  if (action === 'verify') {
    b.status = 'verified';
    b.verifiedBy = ROLES[role].name;
    b.verifiedAt = new Date().toISOString().slice(0,10);
    if (remarks) b.verifyRemarks = remarks;
    showToast('✓','Bill verified — awaiting GM approval');
  } else if (action === 'approve') {
    b.status = 'approved';
    b.approvedBy = ROLES[role].name;
    b.approvedAt = new Date().toISOString().slice(0,10);
    if (remarks) b.approveRemarks = remarks;
    showToast('✅','Bill approved & certified — quantity deducted from ledger');
  } else {
    b.status = 'rejected';
    b.rejectedBy = ROLES[role].name;
    b.rejectReason = remarks;
    showToast('✗','Bill rejected');
  }
  closeModal('approveModal');
  renderAll();
}

function openModal(id)  { document.getElementById(id).classList.add('open'); }
function closeModal(id) { document.getElementById(id).classList.remove('open'); }

function showProjectModal() {
  // Simple inline — could be expanded
  const name = prompt('Project name:', document.getElementById('activeProject').textContent);
  if (name) document.getElementById('activeProject').textContent = name;
}

let toastTimer;
function showToast(icon, msg) {
  const t = document.getElementById('toast');
  document.getElementById('toastIcon').textContent = icon;
  document.getElementById('toastMsg').textContent = msg;
  t.classList.add('show');
  clearTimeout(toastTimer);
  toastTimer = setTimeout(() => t.classList.remove('show'), 3000);
}

  renderWorks();
  renderBills();
  renderApprovals();
  renderLedger();
}
